radif_reg_bank

Configurable RadIF register bank with software-visible storage. Provides ready/valid read and write responses for a parameterized set of memory-mapped control/status registers.

Use Cases

  • Memory-mapped control/status integration with software-visible register maps.

HDL Example

entity radif_reg_bank is
  generic (
    -- Sets the bit width for DATA WIDTH values carried by this module.
    DATA_WIDTH         : integer := 32;
    -- Sets the bit width for ADDR WIDTH values carried by this module.
    ADDR_WIDTH         : integer := 16;
    -- Configures READ ONLY REG COUNT for this instance.
    READ_ONLY_REG_COUNT  : integer := 8;
    -- Configures READ WRITE REG COUNT for this instance.
    READ_WRITE_REG_COUNT : integer := 64;
    -- Selects the vendor-specific implementation path, usually XILINX for DSP48/XPM-backed builds or generic for portable RTL.
    VENDOR_TAG         : string  := "XILINX";
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    PRODUCT_SERIES_TAG : string  := "GENERIC"
  );
  port (
    -- Clock for the associated synchronous logic and handshake domain.
    clk        : in  std_logic;
    -- Active-low reset for this clock domain.
    rstn       : in  std_logic;
    -- Wr addr interface signal.
    wr_addr    : in  std_logic_vector(ADDR_WIDTH - 1 downto 0);
    -- Readback control or data signal for host-visible captured state.
    rd_addr    : in  std_logic_vector(ADDR_WIDTH - 1 downto 0);
    -- Wr en interface signal.
    wr_en      : in  std_logic;
    -- Readback control or data signal for host-visible captured state.
    rd_en      : in  std_logic;
    -- Data in interface signal.
    data_in    : in  std_logic_vector(DATA_WIDTH - 1 downto 0);
    -- Data out interface signal.
    data_out   : out std_logic_vector(DATA_WIDTH - 1 downto 0);
    -- Input read only regs i signal for this module.
    read_only_regs_i  : in  radif_reg_array_t(0 to READ_ONLY_REG_COUNT - 1)(DATA_WIDTH - 1 downto 0);
    -- Output read write regs o signal generated by this module.
    read_write_regs_o : out radif_reg_array_t(0 to READ_WRITE_REG_COUNT - 1)(DATA_WIDTH - 1 downto 0);
    -- Wr rdy interface signal.
    wr_rdy     : out std_logic;
    -- Readback control or data signal for host-visible captured state.
    rd_rdy     : out std_logic;
    -- Wr valid interface signal.
    wr_valid   : out std_logic;
    -- Readback control or data signal for host-visible captured state.
    rd_valid   : out std_logic;
    -- Error interface signal.
    error      : out std_logic
  );
end entity;
