raddsp_xilinx_dsp48_mul

DSP48-backed signed multiplier wrapper for Xilinx FPGA targets. Standardizes valid/last pipelining around vendor DSP blocks so higher-level RadDSP cores can share one multiplication interface.

Use Cases

  • Reusable FPGA DSP pipelines that need fixed-point, timing-aware implementation.

HDL Example

entity raddsp_xilinx_dsp48_mul is
  generic (
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    DEVICE_FAMILY : string := "7series";
    -- Sets the bit width for A WIDTH values carried by this module.
    A_WIDTH       : positive := 16;
    -- Sets the bit width for B WIDTH values carried by this module.
    B_WIDTH       : positive := 18
  );
  port (
    -- Clock for the associated synchronous logic and handshake domain.
    clk        : in  std_logic;
    -- Active-high synchronous reset for this clock domain.
    rst        : in  std_logic;
    -- Input valid qualifier for this operation or sideband channel.
    valid_i    : in  std_logic;
    -- Input subtract i signal for this module.
    subtract_i : in  std_logic;
    -- Input last i signal for this module.
    last_i     : in  std_logic;
    -- Input a i signal for this module.
    a_i        : in  signed(A_WIDTH - 1 downto 0);
    -- Input b i signal for this module.
    b_i        : in  signed(B_WIDTH - 1 downto 0);
    -- Output valid qualifier aligned with the produced result.
    valid_o    : out std_logic;
    -- Output subtract o signal generated by this module.
    subtract_o : out std_logic;
    -- Output last o signal generated by this module.
    last_o     : out std_logic;
    -- Output p o signal generated by this module.
    p_o        : out signed(47 downto 0)
  );
end entity;
