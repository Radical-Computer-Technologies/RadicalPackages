radila_v1_0

Vivado-packaged AXI wrapper around the RadILA capture core. Maps AXI register transactions onto analyzer control, trigger, status, and capture-buffer readback signals.

Use Cases

  • Debug and observability hooks for generated FPGA systems.

entity radila_v1_0 is
  generic (
    -- Sets the bit width for C S00 AXI DATA WIDTH values carried by this module.
    C_S00_AXI_DATA_WIDTH : integer := 32;
    -- Sets the bit width for C S00 AXI ADDR WIDTH values carried by this module.
    C_S00_AXI_ADDR_WIDTH : integer := 6;
    -- Sets the bit width for SAMPLE WIDTH values carried by this module.
    SAMPLE_WIDTH         : integer := 32;
    -- Sets the bit width for EVENT WIDTH values carried by this module.
    EVENT_WIDTH          : integer := 8;
    -- Sets the storage depth, frame length, or number of buffered samples used internally.
    DEPTH                : integer := 1024;
    -- Sets the bit width for ADDR WIDTH values carried by this module.
    ADDR_WIDTH           : integer := 10;
    -- Sets the number of parallel sample lanes processed per handshake beat.
    CMD_LANES            : integer := 4;
    -- Selects the vendor-specific implementation path, usually XILINX for DSP48/XPM-backed builds or generic for portable RTL.
    VENDOR_TAG           : string  := "XILINX";
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    PRODUCT_SERIES_TAG   : string  := "7SERIES";
    -- Configures G DEBUG BUS for this instance.
    G_DEBUG_BUS          : string  := "AXI_LITE"
  );
  port (
    -- Input sample vector captured or processed by the datapath.
    sample_i       : in  std_logic_vector(SAMPLE_WIDTH - 1 downto 0);
    -- Event or trigger metadata associated with the current sample stream.
    event_i        : in  std_logic_vector(EVENT_WIDTH - 1 downto 0);
    -- Output irq o signal generated by this module.
    irq_o          : out std_logic;
    -- S00 axi aclk interface signal.
    s00_axi_aclk   : in  std_logic;
    -- S00 axi aresetn interface signal.
    s00_axi_aresetn: in  std_logic;
    -- S00 axi awaddr interface signal.
    s00_axi_awaddr : in  std_logic_vector(C_S00_AXI_ADDR_WIDTH - 1 downto 0);
    -- S00 axi awprot interface signal.
    s00_axi_awprot : in  std_logic_vector(2 downto 0);
    -- S00 axi awvalid interface signal.
    s00_axi_awvalid: in  std_logic;
    -- S00 axi awready interface signal.
    s00_axi_awready: out std_logic;
    -- S00 axi wdata interface signal.
    s00_axi_wdata  : in  std_logic_vector(C_S00_AXI_DATA_WIDTH - 1 downto 0);
    -- S00 axi wstrb interface signal.
    s00_axi_wstrb  : in  std_logic_vector((C_S00_AXI_DATA_WIDTH / 8) - 1 downto 0);
    -- S00 axi wvalid interface signal.
    s00_axi_wvalid : in  std_logic;
    -- S00 axi wready interface signal.
    s00_axi_wready : out std_logic;
    -- S00 axi bresp interface signal.
    s00_axi_bresp  : out std_logic_vector(1 downto 0);
    -- S00 axi bvalid interface signal.
    s00_axi_bvalid : out std_logic;
    -- S00 axi bready interface signal.
    s00_axi_bready : in  std_logic;
    -- S00 axi araddr interface signal.
    s00_axi_araddr : in  std_logic_vector(C_S00_AXI_ADDR_WIDTH - 1 downto 0);
    -- S00 axi arprot interface signal.
    s00_axi_arprot : in  std_logic_vector(2 downto 0);
    -- S00 axi arvalid interface signal.
    s00_axi_arvalid: in  std_logic;
    -- S00 axi arready interface signal.
    s00_axi_arready: out std_logic;
    -- S00 axi rdata interface signal.
    s00_axi_rdata  : out std_logic_vector(C_S00_AXI_DATA_WIDTH - 1 downto 0);
    -- S00 axi rresp interface signal.
    s00_axi_rresp  : out std_logic_vector(1 downto 0);
    -- S00 axi rvalid interface signal.
    s00_axi_rvalid : out std_logic;
    -- S00 axi rready interface signal.
    s00_axi_rready : in  std_logic
  );
end entity;
