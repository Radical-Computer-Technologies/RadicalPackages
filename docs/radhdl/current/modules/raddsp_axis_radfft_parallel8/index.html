raddsp_axis_radfft_parallel8

Eight-lane parallel AXI-stream RadFFT engine. Processes multiple samples per cycle for high-throughput spectral pipelines on larger FPGA devices.

Use Cases

  • AXI-Stream datapath integration where module boundaries need explicit ready/valid behavior.
  • Reusable FPGA DSP pipelines that need fixed-point, timing-aware implementation.
  • Memory-mapped control/status integration with software-visible register maps.

entity raddsp_axis_radfft_parallel8 is
  generic (
    -- Selects the vendor-specific implementation path, usually XILINX for DSP48/XPM-backed builds or generic for portable RTL.
    VENDOR               : string := "xilinx";
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    DEVICE_FAMILY        : string := "ultrascaleplus";
    -- Sets the transform, frame, or vector size used by the datapath.
    G_POINTS             : positive := 1024;
    -- Sets the transform, frame, or vector size used by the datapath.
    G_MAX_POINTS         : positive := 1024;
    -- Sets the transform, frame, or vector size used by the datapath.
    G_INVERSE_FFT        : boolean := false;
    -- Sets the bit width for G INPUT WIDTH values carried by this module.
    G_INPUT_WIDTH        : positive := 32;
    -- Sets the bit width for G TWIDDLE WIDTH values carried by this module.
    G_TWIDDLE_WIDTH      : positive := 16;
    -- Sets the bit width for G OUTPUT WIDTH values carried by this module.
    G_OUTPUT_WIDTH       : positive := 32;
    -- Configures G SCALE EACH STAGE for this instance.
    G_SCALE_EACH_STAGE   : boolean := true;
    -- Sets the number of parallel sample lanes processed per handshake beat.
    G_MULTIPLIER_LANES   : positive := 8
  );
  port (
    -- Clock for the associated synchronous logic and handshake domain.
    clk            : in  std_logic;

    s_axis_tready  : out std_logic;
    -- Input AXI-stream payload containing packed sample or feature lanes.
    s_axis_tdata   : in  std_logic_vector((2 * G_INPUT_WIDTH) - 1 downto 0);
    -- Input AXI-stream frame marker for the final beat of a frame.
    s_axis_tlast   : in  std_logic;

    m_axis_tvalid  : out std_logic;
    -- Output AXI-stream ready input from the downstream block.
    m_axis_tready  : in  std_logic;
    -- Output AXI-stream payload containing packed processed result lanes.
    m_axis_tdata   : out std_logic_vector((2 * G_OUTPUT_WIDTH) - 1 downto 0);
    -- Output AXI-stream frame marker aligned with the processed result beat.
    m_axis_tlast   : out std_logic;
    -- Input runtime inverse i signal for this module.
    runtime_inverse_i : in std_logic := '0';
    -- Input active point log2 i signal for this module.
    active_point_log2_i : in std_logic_vector(7 downto 0) := (others => '0');
    -- Input active radix4 i signal for this module.
    active_radix4_i     : in std_logic := '0';
    -- Input active input width i signal for this module.
    active_input_width_i : in std_logic_vector(7 downto 0) := (others => '0');
    -- Input active output width i signal for this module.
    active_output_width_i : in std_logic_vector(7 downto 0) := (others => '0');
    -- Output twiddle addr o signal generated by this module.
    twiddle_addr_o : out std_logic_vector(31 downto 0);
    -- Input twiddle re i signal for this module.
    twiddle_re_i   : in  std_logic_vector(G_TWIDDLE_WIDTH - 1 downto 0);
    -- Input twiddle im i signal for this module.
    twiddle_im_i   : in  std_logic_vector(G_TWIDDLE_WIDTH - 1 downto 0);
