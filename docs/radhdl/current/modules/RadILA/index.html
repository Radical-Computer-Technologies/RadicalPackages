RadILA

Integrated logic analyzer capture core for RadHDL debug builds. Samples an application bus in the sample clock domain, captures event metadata, and exposes captured data through an AXI-facing read path.

Use Cases

  • Debug and observability hooks for generated FPGA systems.

HDL Example

entity RadILA is
  generic (
    -- Sets the bit width for SAMPLE WIDTH values carried by this module.
    SAMPLE_WIDTH : integer := 32;
    -- Sets the bit width for EVENT WIDTH values carried by this module.
    EVENT_WIDTH  : integer := 8;
    -- Sets the storage depth, frame length, or number of buffered samples used internally.
    DEPTH        : integer := 1024;
    -- Sets the bit width for ADDR WIDTH values carried by this module.
    ADDR_WIDTH   : integer := 10;
    -- Sets the number of parallel sample lanes processed per handshake beat.
    CMD_LANES    : integer := 4;
    -- Selects the vendor-specific implementation path, usually XILINX for DSP48/XPM-backed builds or generic for portable RTL.
    VENDOR_TAG   : string  := "XILINX";
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    PRODUCT_SERIES_TAG : string := "7SERIES"
  );
  port (
    -- Clock for the associated synchronous logic and handshake domain.
    sample_clk       : in  std_logic;
    -- Active-low reset for this clock domain.
    sample_rstn      : in  std_logic;
    -- Clock for the associated synchronous logic and handshake domain.
    axi_clk          : in  std_logic;
    -- Active-low reset for this clock domain.
    axi_rstn         : in  std_logic;
    -- Input sample vector captured or processed by the datapath.
    sample_i         : in  std_logic_vector(SAMPLE_WIDTH - 1 downto 0);
    -- Event or trigger metadata associated with the current sample stream.
    event_i          : in  std_logic_vector(EVENT_WIDTH - 1 downto 0);
    -- Command/control input used to configure or trigger the block.
    cmd_data_i       : in  std_logic_vector(CMD_LANES - 1 downto 0);
    -- Command/control input used to configure or trigger the block.
    cmd_toggle_i     : in  std_logic;
    -- Command/control acknowledgement or status output.
    cmd_ack_toggle_o : out std_logic;
    -- Readback control or data signal for host-visible captured state.
    rd_index_i       : in  unsigned(ADDR_WIDTH - 1 downto 0);
    -- Readback control or data signal for host-visible captured state.
    rd_data_o        : out std_logic_vector(SAMPLE_WIDTH - 1 downto 0);
    -- Output sample vector or current sample status from the datapath.
    sample_now_o     : out std_logic_vector(SAMPLE_WIDTH - 1 downto 0);
    -- Event or trigger metadata associated with the current sample stream.
    event_now_o      : out std_logic_vector(EVENT_WIDTH - 1 downto 0);
    -- Output armed o signal generated by this module.
    armed_o          : out std_logic;
    -- Output capturing o signal generated by this module.
    capturing_o      : out std_logic;
    -- Output done o signal generated by this module.
    done_o           : out std_logic;
    -- Output overflow o signal generated by this module.
    overflow_o       : out std_logic;
    -- Output count o signal generated by this module.
    count_o          : out unsigned(ADDR_WIDTH downto 0)
  );
end entity;
