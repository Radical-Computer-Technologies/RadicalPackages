radhdl_dsp_axi_smoke_top

Integration top level for the RadHDL DSP AXI smoke design. Connects RadDSP streaming and register-interface blocks into a board-testable reference design with address-map artifacts.

Use Cases

  • Memory-mapped control/status integration with software-visible register maps.

entity radhdl_dsp_axi_smoke_top is
  generic (
    -- Selects the vendor-specific implementation path, usually XILINX for DSP48/XPM-backed builds or generic for portable RTL.
    VENDOR_TAG         : string := "XILINX";
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    PRODUCT_SERIES_TAG : string := "GENERIC";
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    DSP48_FAMILY       : string := "7series";
    -- Sets the number of parallel sample lanes processed per handshake beat.
    FIR_DSP_LANES      : positive := 1;
    -- Sets the number of parallel sample lanes processed per handshake beat.
    BIQUAD_DSP_LANES   : positive := 1;
    -- Sets the bit width for DMA DATA WIDTH values carried by this module.
    DMA_DATA_WIDTH      : positive := 64
  );
  port (
    -- Aclk interface signal.
    aclk           : in  std_logic;
    -- Aresetn interface signal.
    aresetn        : in  std_logic;
    -- AXI4-Lite slave-channel signal used for host register access.
    s_axi_awaddr   : in  std_logic_vector(15 downto 0);
