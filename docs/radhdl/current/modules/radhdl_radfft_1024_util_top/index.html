radhdl_radfft_1024_util_top

Synthesis-report top level for RadFFT 1024-point utilization studies. Instantiates the target FFT core configuration with minimal surrounding logic for repeatable timing and resource reports.

Use Cases

  • Reusable FPGA DSP pipelines that need fixed-point, timing-aware implementation.

HDL Example

entity radhdl_radfft_1024_util_top is
  generic (
    -- Identifies the target FPGA family so wrappers can choose the correct primitive or conservative portable behavior.
    DEVICE_FAMILY    : string := "ultrascaleplus";
    -- Sets the transform, frame, or vector size used by the datapath.
    FFT_RADIX        : positive := 4;
    -- Sets the transform, frame, or vector size used by the datapath.
    FFT_MEMORY_STYLE : string := "block";
    -- Sets the transform, frame, or vector size used by the datapath.
    FFT_TWIDDLE_MEMORY_STYLE : string := "block";
    -- Sets the transform, frame, or vector size used by the datapath.
    FFT_TWIDDLE_INIT_FILE : string := "none";
    -- Sets the transform, frame, or vector size used by the datapath.
    FFT_WORD_SAMPLES : positive := 4;
    -- Sets the bit width for INPUT WIDTH values carried by this module.
    INPUT_WIDTH      : positive := 16;
    -- Sets the bit width for TWIDDLE WIDTH values carried by this module.
    TWIDDLE_WIDTH    : positive := 16;
    -- Sets the bit width for OUTPUT WIDTH values carried by this module.
    OUTPUT_WIDTH     : positive := 24
  );
  port (
    -- Clock for the associated synchronous logic and handshake domain.
    clk      : in  std_logic;
    -- Active-low reset for this clock domain.
    rstn     : in  std_logic;
    -- Output digest o signal generated by this module.
    digest_o : out std_logic_vector(31 downto 0)
  );
end entity;
