settings.json

{
  "radbuild_schema": "2.0",
  "radbuild_version": "0.2.1",
  "project_name": "radpm-demo",
  "systems": [
    {
      "id": "radix-x86",
      "type": "os.radix",
      "provider": "radix-os",
      "depends_on": [],
      "config": {
        "ui_profile": "terminal",
        "feature_chunks": ["kernel.core", "userspace.rash"],
        "radpm": {
          "repository": "../RadicalPackages",
          "suite": "experimental",
          "packages": ["radix-core"]
        }
      }
    }
  ],
  "connections": [],
  "artifacts": [],
  "deploy": [],
  "packages": {}
}

Commands

radbuild menuconfig --settings settings.json --enable-package radlib-ui --resolve-packages --json

Expected selected package order:

radix-core
radlib-runtime
radlib-ui

Expected lock path:

.radmeta/radix-os/radix-x86/radpm/radpm-lock.json
