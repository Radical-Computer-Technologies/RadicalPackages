Example: FPGA + PetaLinux MPSoC Project

settings.json

{
  "radbuild_schema": "2.0",
  "radbuild_version": "0.2.1",
  "project_name": "mpsoc-demo",
  "systems": [
    {
      "id": "vivado",
      "type": "fpga.vivado",
      "provider": "vivado",
      "depends_on": [],
      "config": {
        "project_name": "mpsoc_hw",
        "part_number": "xczu3eg-sbva484-1-e",
        "version": "2024.1"
      }
    },
    {
      "id": "petalinux",
      "type": "linux.petalinux",
      "provider": "petalinux",
      "depends_on": ["vivado"],
      "config": {
        "project_name": "mpsoc_linux",
        "template": "zynqMP",
        "version": "2024.1"
      }
    }
  ],
  "connections": [],
  "artifacts": [],
  "deploy": [],
  "packages": {}
}

Commands

radbuild project create ./workspace --non-interactive --template fpga-mpsoc-petalinux --project-name mpsoc-demo
radbuild graph --settings settings.json --json
radbuild generate --settings settings.json
radbuild build all --settings settings.json --dry-run --json-events

Expected Output

{"event":"system_planned","system":"vivado","type":"fpga.vivado"}
{"event":"system_planned","system":"petalinux","type":"linux.petalinux"}
